```
    const arr = [-10, 21, 3, 3, 5, 2, 8, 1, 99, 2, 9, 4, 7, 6, 20]

    function quickSort(array) {
      const mid = array[0]
      const left = []
      const right = []
      if (array.length <= 1) {
        return array
      }
      array.splice(0, 1)
      for (let i = 0; i < array.length; i++) {
        if (array[i] < mid) {
          left.push(array[i])
        } else {
          right.push(array[i])
        }
      }
      return [...quickSort(left), mid, ...quickSort(right)]
    }

    console.log(quickSort(arr));
```
