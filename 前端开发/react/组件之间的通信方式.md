# react组件之间的通信方式

- props
这种比较简单，父组件传参数，自组件直接引用

```
// Parent

render() {
   return <Child msg={this.state.msg} />;
 }


// Child

render() {
  return <p>{this.props.msg}</p>
}
```


- 自组件执行父组件的方法，也是通过props传递，只不过这次传递的是个function


```
// Parent

transferMsg(self: Parent, other: any) {
  console.log(self);
  self.setState({
    test: other
  });
}
render() {
   return <Child msg={this.state.msg} fun={this.transferMsg.bind(this, this)}/>;
 }


// Child

componentDidMount() {
    this.props.fun('update')
  }
render() {
  return <p>{this.props.msg}</p>
}
```

- 使用eventBus,原理就是一个观察者模式，一个组件负责监听事件变化，一个组件执行方法

```
// component1
eventProxy.on('msg', (msg: string) => {
  console.log(msg);
});

// component2
eventProxy.trigger('msg', 'end');

```
```
// eventProxy.js
'use strict';
const eventProxy = {
  onObj: {},
  oneObj: {},
  on: function(key, fn) {
    if(this.onObj[key] === undefined) {
      this.onObj[key] = [];
    }

    this.onObj[key].push(fn);
  },
  one: function(key, fn) {
    if(this.oneObj[key] === undefined) {
      this.oneObj[key] = [];
    }

    this.oneObj[key].push(fn);
  },
  off: function(key) {
    this.onObj[key] = [];
    this.oneObj[key] = [];
  },
  trigger: function() {
    let key, args;
    if(arguments.length == 0) {
      return false;
    }
    key = arguments[0];
    args = [].concat(Array.prototype.slice.call(arguments, 1));

    if(this.onObj[key] !== undefined
      && this.onObj[key].length > 0) {
      for(let i in this.onObj[key]) {
        this.onObj[key][i].apply(null, args);
      }
    }
    if(this.oneObj[key] !== undefined
      && this.oneObj[key].length > 0) {
      for(let i in this.oneObj[key]) {
        this.oneObj[key][i].apply(null, args);
        this.oneObj[key][i] = undefined;
      }
      this.oneObj[key] = [];
    }
  }
};

export default eventProxy;
```

- redux

使用redux状态管理来进行组件之间的通信

```
// store.ts

import { createStore } from 'redux'

function counter(state = 0, action: any) {
  switch (action.type) {
    case 'INCREMENT':
      return state + 1
    case 'DECREMENT':
      return state - 1
    default:
      return state
  }
}

const store = createStore(counter)

store.subscribe(() => console.log(store.getState()))

export { store }
```

```
// component

store.dispatch({ type: 'DECREMENT' })
```
